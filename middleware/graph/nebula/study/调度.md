# 调度

```nGQL
point
    point_id
    uid
    point_name
    longitude
    latitude
    altitude
    point_type
    status

CREATE tag `point` (`point_id` int NOT NULL  COMMENT "点ID", `uid` string NOT NULL  COMMENT "点UID", `point_name` string NOT NULL  COMMENT "点名称", `longitude` string NOT NULL  COMMENT "经度", `latitude` string NOT NULL  COMMENT "纬度", `altitude` string NOT NULL  COMMENT "海拔", `point_type` int8 NOT NULL  COMMENT "点类型", `status` int8 NOT NULL  COMMENT "点状态")  COMMENT = "点"

```

```nGQL
point_map
    id
    begin_point_id
    end_point_id
    distance
    direct

CREATE edge `point_map` (`id` int NOT NULL  COMMENT "ID", `begin_point_id` int NOT NULL  COMMENT "起点ID", `end_point_id` int NOT NULL  COMMENT "终点ID", `distance` string NOT NULL  COMMENT "距离", `direct` string NOT NULL  COMMENT "方向角度")  COMMENT = "点与点的关联"
```
