## 2 打印

```go
func main(){
    // 10, 2, 8, 16 进制

    age := 15

    fmt.Printf("%d\n", age)
    fmt.Printf("%b\n", age)
    fmt.Printf("%o\n", age)
    fmt.Printf("%x\n", age)

    name := "pandora"

    // %T type
    fmt.Printf("%T\n", name)

    // %v value
    fmt.Printf("%v\n", name)

    // "pandora"
    fmt.Printf("%#v\n", name)

    // %s 字符串
    fmt.Printf("%s\n", name)

    // %c 输出单个字符
    fmt.Printf("$c\n", 'a')

    // %u 无符号十进制数
}
```
