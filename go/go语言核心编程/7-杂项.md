## 2 打印

```go
func main(){
    // 10, 2, 8, 16 进制

    age := 15

    fmt.Printf("%d\n", age)
    fmt.Printf("%b\n", age)
    fmt.Printf("%o\n", age)
    fmt.Printf("%x\n", age)

    name := "pandora"

    fmt.Printf("%T\n", name)
    fmt.Printf("%v\n", name)
    fmt.Printf("%#v\n", name)
    fmt.Printf("%s\n", name)
}
```
